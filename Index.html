üí≥ Accept Crypto Payments on Your Website

Add direct wallet-to-wallet payments without middlemen. Example button (MetaMask + Ethers.js):

<button onclick="pay()">Send 0.01 ETH</button>
<script src="https://cdn.jsdelivr.net/npm/ethers@6.0.0/dist/ethers.min.js"></script>
<script>
async function pay() {
  if (!window.ethereum) {
    alert("Please install MetaMask first!");
    return;
  }
  const provider = new ethers.BrowserProvider(window.ethereum);
  const signer = await provider.getSigner();
  const tx = await signer.sendTransaction({
    to: "0xYourWalletAddressHere",
    value: ethers.parseEther("0.01")
  });
  alert("Payment sent! Tx: " + tx.hash);
}
</script>

üìú Create a Smart Contract for Your Business

Simple Solidity contract for escrow-style payments between client and business:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract BusinessPayment {
    address public owner;
    address public client;
    uint public amount;
    bool public serviceCompleted;

    constructor(address _client, uint _amount) {
        owner = msg.sender;
        client = _client;
        amount = _amount;
    }

    function deposit() public payable {
        require(msg.sender == client, "Only client can deposit");
        require(msg.value == amount, "Incorrect amount");
    }

    function markServiceCompleted() public {
        require(msg.sender == owner, "Only owner can confirm");
        serviceCompleted = true;
    }

    function releaseFunds() public {
        require(serviceCompleted, "Service not completed yet");
        payable(owner).transfer(address(this).balance);
    }

    function refundClient() public {
        require(!serviceCompleted, "Already completed");
        payable(client).transfer(address(this).balance);
    }
}

Deploy this on Remix IDE with MetaMask connected to Ethereum or Polygon.
